def index(request):

    query = request.GET.get('q', '').strip()

    categories = request.GET.getlist('categories')

    sort = request.GET.get('sort', 'hotness_desc')

    

    clubs = Club.objects.annotate(avg_score=Avg('ratings__score'), member_count=Count('members'))

    

    # 脙娄脗聬脗聹脙搂脗麓脗垄

    if query:

        clubs = clubs.filter(Q(name__icontains=query) | Q(description__icontains=query) | Q(category__icontains=query))

    

    # 脙搂脗颅脗聸脙漏脗聙?

    if categories:

        clubs = clubs.filter(category__in=categories)

    

    # 脙篓脗庐脗隆脙搂脗庐脗聴脙楼脗聬脗聞脙楼脗颅脗聴脙娄脗庐脗碌脙娄脗聨脗聮脙楼脗潞脗聫脙炉脗录脗聢脙陇脗赂脗聧脙娄脗聰脗鹿脙楼脗聫脗聵queryset脙搂脗禄脗聯脙娄脗聻脗聞脙炉脗录脗聦脙陇脗驴脗聺脙篓脗炉脗聛脙楼脗聢脗聠脙漏脗隆脗碌脙娄脗颅脗拢脙楼脗赂脗赂脙炉脗录脗聣

    hotness_map = {}

    for club in clubs:

        hotness_map[club.pk] = club.get_hotness()



    if sort == 'created_asc':

        clubs = clubs.order_by('created_at')

    elif sort == 'created_desc':

        clubs = clubs.order_by('-created_at')

    elif sort == 'members_asc':

        clubs = clubs.order_by('member_count')

    elif sort == 'members_desc':

        clubs = clubs.order_by('-member_count')

    elif sort == 'hotness_asc':

        clubs = list(clubs)

        clubs.sort(key=lambda c: hotness_map[c.pk])

    elif sort == 'hotness_desc':

        clubs = list(clubs)

        clubs.sort(key=lambda c: hotness_map[c.pk], reverse=True)

    else:

        clubs = clubs.order_by('-created_at')

    

    paginator = Paginator(clubs, 8)

    page_number = request.GET.get('page')

    page_obj = paginator.get_page(page_number)

    

    joined_clubs = list(request.user.joined_clubs.all()) if request.user.is_authenticated else []

    joined_club_ids = [c.id for c in joined_clubs]

    recommended_clubs = Club.objects.annotate(avg_score=Avg('ratings__score')).order_by('-avg_score')

    if request.user.is_authenticated:

        recommended_clubs = recommended_clubs.exclude(members=request.user)[:4]

    else:

        recommended_clubs = recommended_clubs[:4]

    

    return render(request, 'clubs/index.html', {

        'page_obj': page_obj,

        'query': query,

        'categories': categories,

        'sort': sort,

        'joined_clubs': joined_clubs,

        'joined_club_ids': joined_club_ids,

        'recommended_clubs': recommended_clubs,

        'category_choices': Club.CATEGORY_CHOICES,

    })

def club_detail(request, club_id):

    club = get_object_or_404(Club, id=club_id)

    posts = club.posts.select_related('author').prefetch_related('likes', 'comments__author').all()

    paginator = Paginator(posts, 5)

    page_number = request.GET.get('page')

    posts_page = paginator.get_page(page_number)

    is_member = request.user.is_authenticated and club.members.filter(id=request.user.id).exists()

    is_admin = request.user.is_authenticated and club.is_admin(request.user)

    is_owner = request.user.is_authenticated and club.creator == request.user

    memberships = club.membership_set.select_related('user').all()

    post_form = PostForm() if is_member else None

    rating = None

    rating_form = None

    if request.user.is_authenticated:

        rating = ClubRating.objects.filter(club=club, user=request.user).first()

        rating_form = ClubRatingForm(instance=rating, club=club, user=request.user)

    cross_events = club.cross_events.filter(is_public=True)

    # 脙篓脗聨脗路脙楼脗聫脗聳脙搂脗聰脗篓脙娄脗聢脗路脙楼脗路脗虏脙搂脗聜脗鹿脙篓脗碌脗聻脙搂脗職脗聞脙楼脗聤脗篓脙娄脗聙脗聛ID脙楼脗聢脗聴脙篓脗隆脗篓

    liked_post_ids = []

    if request.user.is_authenticated:

        liked_post_ids = list(PostLike.objects.filter(user=request.user, post__club=club).values_list('post_id', flat=True))

    return render(

        request,

        'clubs/club_detail.html',

        {

            'club': club,

            'posts': posts_page,

            'is_member': is_member,

            'is_admin': is_admin,

            'is_owner': is_owner,

            'memberships': memberships,

            'post_form': post_form,

            'rating_form': rating_form,

            'ratings': club.ratings.select_related('user').all(),

            'knowledge_categories': club.knowledge_categories.prefetch_related('items').all(),

            'liked_post_ids': liked_post_ids,

            'cross_events': cross_events,

        },

    )

def events_list(request):

    query = request.GET.get('q', '').strip()

    categories = request.GET.getlist('categories')

    scopes = request.GET.getlist('scopes')

    sort = request.GET.get('sort', 'created_desc')



    events = CrossSchoolEvent.objects.filter(is_public=True)



    if query:

        events = events.filter(

            Q(title__icontains=query)

            | Q(description__icontains=query)

            | Q(partner_school__icontains=query)

            | Q(club__name__icontains=query)

        )



    if categories:

        events = events.filter(category__in=categories)



    if scopes:

        events = events.filter(scope__in=scopes)



    if sort == 'start_asc':

        events = events.order_by('start_date')

    elif sort == 'start_desc':

        events = events.order_by('-start_date')

    elif sort == 'participants_asc':

        events = events.annotate(participant_count=Count('participants')).order_by('participant_count')

    elif sort == 'participants_desc':

        events = events.annotate(participant_count=Count('participants')).order_by('-participant_count')

    elif sort == 'created_asc':

        events = events.order_by('created_at')

    else:

        events = events.order_by('-created_at')



    paginator = Paginator(events, 8)

    page_number = request.GET.get('page')

    page_obj = paginator.get_page(page_number)



    return render(request, 'clubs/events_list.html', {

        'page_obj': page_obj,

        'query': query,

        'categories': categories,

        'scopes': scopes,

        'sort': sort,

        'category_choices': Club.CATEGORY_CHOICES,

        'scope_choices': CrossSchoolEvent.EVENT_SCOPE_CHOICES,

    })

def event_ical(request, event_id):

    event = get_object_or_404(CrossSchoolEvent, id=event_id, is_public=True)

    dtstamp = datetime.datetime.now().strftime('%Y%m%dT%H%M%SZ')

    dtstart = event.start_date.strftime('%Y%m%d')

    dtend = (event.end_date + datetime.timedelta(days=1)).strftime('%Y%m%d')

    ical = f"""BEGIN:VCALENDAR\nVERSION:2.0\nPRODID:-//CampusClub//Event//CN\nBEGIN:VEVENT\nUID:event-{event.id}@campusclub\nDTSTAMP:{dtstamp}\nDTSTART;VALUE=DATE:{dtstart}\nDTEND;VALUE=DATE:{dtend}\nSUMMARY:{event.title}\nDESCRIPTION:{event.description.replace('\n', ' ')}\nLOCATION:{event.location or ''}\nEND:VEVENT\nEND:VCALENDAR\n"""

    response = HttpResponse(ical, content_type='text/calendar')

    response['Content-Disposition'] = f'attachment; filename=event_{event.id}.ics'

    return response
